# Creando todas las funciones para limpiar Na


```python
def imputation_mode(frame, column_name=None):
    dataframe = frame
    if column_name==None:
        for i in range(183):
            dataframe = dataframe.fillna(dataframe.mode().iloc[i])
    else:
        dataframe[[column_name]] = dataframe[[column_name]].fillna(dataframe.mode().iloc[0])
    return dataframe

def imputation_mean(frame, column_name=None):
    dataframe = frame
    if column_name==None:
        for i in range(7):
            dataframe = dataframe.fillna(dataframe.mean().iloc[i])
    else:
        dataframe[[column_name]] = dataframe[[column_name]].fillna(dataframe[[column_name]].mean())
    return dataframe

def imputation_median(frame, column_name=None):
    dataframe = frame
    if column_name==None:
        for i in range(11):
            dataframe = dataframe.fillna(dataframe.median().iloc[i])
    else:
        dataframe[[column_name]] = dataframe[[column_name]].fillna(dataframe[[column_name]].median())
    return dataframe

def listwise_deletion(dataframe):
    dataframe = dataframe.dropna()
    return dataframe

def listwise_deletion_column(dataframe, column):
    dataframe = dataframe.dropna(subset=column)
    return dataframe

def pairwise_deletion(dataframe, columns): #columns es un arreglo
    dataframe = dataframe.dropna(subset=columns)
    return dataframe

def outliers_standard(dataframe, column, zscore):
    dataframe = dataframe[(np.abs(stats.zscore(dataframe[[column]])) < zscore).all(axis=1)]
    return dataframe

```
